题目一-题目大意

模拟竖式乘法的过程并输出。

考察算法

竖式运算的格式，模拟。

算法一

思路：

模拟，直接输出，不判断计算出的两个数字的相对位数，输出不带格式。

时间复杂度分析：

O(1)。

期望得分：

25分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int main()
{
	int x,y;
	scanf("%d%d",&x,&y);
	if (y <= 10)
	{
		int sum = x * y;
		printf("%d\n* %d\n",x,y);
		puts("----");
		printf("%d\n",sum);
	}
	else
	{
		int sum1 = x * (y % 10);
		int sum2 = x * (y / 10);
		int sum = sum1 + sum2 * 10;
		printf("%d\n* %d\n",x,y);
		puts("----");
		printf("%d\n%d\n",sum1,sum2);
		puts("----");
		printf("%d\n",sum);
	}
	return 0;
}

算法二

思路：

判断第一个数字乘第二个数字的十位的位数，从而对输出的第一个数字乘第二个数字的个位的结果控制位数。

时间复杂度分析：

O(1)。

期望得分：

100分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int cal(int k)
{
	if (k == 0) return 1;
	int cnt = 0;
	while (k)
	{
		cnt++;
		k /= 10;
	}
	return cnt;
}

int main()
{
	int x,y;
	scanf("%d%d",&x,&y);
	if (y <= 10)
	{
		int sum = x * y;
		printf("%d\n* %d\n",x,y);
		puts("----");
		printf("%d\n",sum);
	}
	else
	{
		int sum1 = x * (y % 10);
		int sum2 = x * (y / 10);
		int sum = sum1 + sum2 * 10;
		printf("%d\n* %d\n",x,y);
		puts("----");
		int b2 = cal(sum2*10);
		if (b2 == 2)
		{
			printf("%2d\n",sum1);
		}
		else if (b2 == 3)
		{
			printf("%3d\n",sum1);
		}
		else if (b2 == 4)
		{
			printf("%4d\n",sum1);
		}
		printf("%d\n",sum2);
		puts("----");
		printf("%d\n",sum);
	}
	return 0;
}

题目二-题目大意

给出一个 $N * N$ 的棋盘，再给出一个格子的坐标，输出与这个格子同一行，同一列，在同一对角线上（正对角线和副对角线均要计算）的格子的坐标。

考察算法

模拟。

算法一

思路：

模拟即可。

时间复杂度分析：

O(N)。

期望得分：

100分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
	int n,x,y;
	scanf("%d%d%d",&n,&x,&y);
	for (int i = 1;i <= n;i++)
	{
		printf("(%d,%d)",x,i);
	}
	puts("");
	for (int i = 1;i <= n;i++)
	{
		printf("(%d,%d)",i,y);
	}
	puts("");
	int tx = x,ty = y;
	while (tx > 1 && ty > 1)
	{
		tx--,ty--;
	}
	while (tx <= n && ty <= n)
	{
		printf("(%d,%d)",tx,ty);
		tx++,ty++;
	}
	puts("");
	tx = x,ty = y;
	while (tx < n && ty > 1)
	{
		tx++,ty--;
	}
	while (tx >= 1 && ty <= n)
	{
		printf("(%d,%d)",tx,ty);
		tx--,ty++;
	}
	puts("");
	return 0;
}

题目三-题目大意

输出一个字符串$s$，并对其进行一次操作。

操作有3种：

D a：表示删除s中出现的第一个a（如果存在）；

I a1 a2：表示把a2插入到s中第一个a1之前，（如果a1存在）；

R a1 a2：表示把s中每一个a1用a2替换（如果存在）。

考察算法

模拟，暴力，字符串基本操作。

算法一

思路：

不进行实际的替换，将删除的字符忽略，将插入的直接输出，将替换的直接输出。

时间复杂度分析：

$O(|s|)$，$|s|$代表字符串的长度。

期望得分：

100分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int main()
{
	char s[50];
	gets(s);
	int sz = strlen(s);
	char op;
	scanf(" %c",&op);
	if (op == 'D')
	{
		char ch;
		scanf(" %c",&ch);
		int pos = -1;
		for (int i = 0;i < sz;i++)
		{
			if (s[i] == ch)
			{
				pos = i;
				break;
			}
		}
		for (int i = 0;i < pos;i++) putchar(s[i]);
		for (int i = pos + 1;i < sz;i++) putchar(s[i]);
	}
	else if (op == 'I')
	{
		char a1,a2;
		scanf(" %c %c",&a1,&a2);
		int pos = -1;
		for (int i = sz - 1;i >= 0;i--)
		{
			if (s[i] == a1)
			{
				pos = i;
				break;
			}
		}
		for (int i = 0;i < pos;i++) putchar(s[i]);
		putchar(a2);
		for (int i = pos;i < sz;i++) putchar(s[i]);
	}
	else
	{
		char a1,a2;
		scanf(" %c %c",&a1,&a2);
		for (int i = 0;i < sz;i++)
		{
			if (s[i] == a1)
			{
				putchar(a2);
			}
			else
			{
				putchar(s[i]);
			}
		}
	}
	puts("");
	return 0;
}

算法二

思路：

模拟，进行实际的替换后再输出。

时间复杂度分析：

$O(|s|)$，$|s|$代表字符串的长度。

期望得分：

100分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
	char s[50];
	gets(s);
	int sz = strlen(s);
	char op;
	scanf(" %c",&op);
	if (op == 'D')
	{
		char ch;
		scanf(" %c",&ch);
		int pos = -1;
		for (int i = 0;i < sz;i++)
		{
			if (s[i] == ch)
			{
				pos = i;
				break;
			}
		}
		for (int i = pos;i < sz;i++)
		{
			s[i] = s[i+1];
		}
		s[sz-1] = 0;
	}
	else if (op == 'I')
	{
		char a1,a2;
		scanf(" %c %c",&a1,&a2);
		int pos = -1;
		for (int i = sz - 1;i >= 0;i--)
		{
			if (s[i] == a1)
			{
				pos = i;
				break;
			}
		}
		if (pos != -1)
		{
			for (int i = sz - 1;i >= pos;i--)
			{
				s[i+1] = s[i];
			}
			s[pos] = a2;
		}
		s[sz+1] = 0;
	}
	else
	{
		char a1,a2;
		scanf(" %c %c",&a1,&a2);
		for (int i = 0;i < sz;i++)
		{
			if (s[i] == a1)
			{
				s[i] = a2;
			}
		}
	}
	puts(s);
	return 0;
}

题目四-题目大意

有$2^n$个球队进行比赛，计划在$2^n - 1$天内完成，每个队伍每天只进行一场比赛，按照字典序输出每天的比赛情况。

考察算法

模拟，按字典序输出。

算法一

思路：

因为n较小，直接模拟即可。

时间复杂度分析：

$O(2^n * n^2)$。

期望得分：

100分。

参考代码：

#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
const int N = 100;

bool vis[N][N];
bool used[N];

int main()
{
	int n;
	scanf("%d",&n);
	int sum = (1 << n);
	int tmp = sum;
	int cnt = 0;

	while (--tmp)
	{
		printf("<%d>",++cnt);
		memset(used,0,sizeof(used));
		bool f = 0;
		while (1)
		{
			bool update = 0;
			for (int i = 1;i <= sum;i++)
			{
				if (!used[i])
				{
					update = 1;
					used[i] = 1;
					for (int j = 1;j <= sum;j++)
					{
						if (!used[j] && !vis[i][j])
						{
							used[j] = 1;
							vis[i][j] = vis[j][i] = 1;
							if (f) printf(" %d-%d",i,j);
							else
							{
								f = 1;
								printf("%d-%d",i,j);
							}
							break;
						}
					}
				}
			}
			if (!update) break;
		}
		puts("");
	}
	return 0;
}
